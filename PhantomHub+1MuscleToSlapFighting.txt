local BASE_VIEWPORT_SIZE = Vector2.new(1600, 720)

local function getResponsiveScale()
    local camera = workspace.CurrentCamera
    local viewportSize = camera and camera.ViewportSize or BASE_VIEWPORT_SIZE
    local scale = math.min(viewportSize.X / BASE_VIEWPORT_SIZE.X, viewportSize.Y / BASE_VIEWPORT_SIZE.Y)
    return math.clamp(scale, 0.85, 2.15)
end

local function rpx(value)
    if value == 0 then
        return 0
    end

    local scale = getResponsiveScale()
    local sign = value < 0 and -1 or 1
    return sign * math.max(1, math.floor(math.abs(value) * scale + 0.5))
end

local savedSettings = {}

}

if type(isfile) == "function" and isfile(fileName) then
    local success, data = pcall(function()
        return HttpService:JSONDecode(readfile(fileName))
    end)
    if success and type(data) == "table" then
        savedSettings = data
    end
end

local function saveSettings()
    if type(writefile) == "function" then
        pcall(function()
            writefile(fileName, HttpService:JSONEncode(savedSettings))
        end)
    end
end

local optionStates = {}

local function createOptionState(name, initialValue)
    if unsavedSettings[name] then
        savedSettings[name] = nil
        optionStates[name] = initialValue
    elseif savedSettings[name] ~= nil then
        optionStates[name] = savedSettings[name]
    else
        optionStates[name] = initialValue
    end
    return function() return optionStates[name] end
end

local function setOptionState(name, value)
    optionStates[name] = value
    if unsavedSettings[name] then
        savedSettings[name] = nil
    else
        savedSettings[name] = value
    end
    saveSettings()
end

local function saveCurrentSettings()
    for name, value in pairs(optionStates) do
        if name ~= "_debounce" and not unsavedSettings[name] then
            savedSettings[name] = value
        end
    end

    saveSettings()
end

local function getCategoryKey(tabName, categoryName)

local function registerElement(tabName, element, elementName)
    if not tabs[tabName] then
        tabs[tabName] = { button = nil, elements = {} }
    end

    table.insert(tabs[tabName].elements, element)
    optionElementsMap[element] = elementName or ""
    optionTabMap[element] = tabName

    local categoryName = currentCategoryByTab[tabName]
    if categoryName then
        local key = getCategoryKey(tabName, categoryName)
        elementCategoryMap[element] = key
        if not categoryChildren[key] then
            categoryChildren[key] = {}
        end
        table.insert(categoryChildren[key], element)
    end
end

		TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{Position = UDim2.new(0.71, rpx(0), 0.8, rpx(0))}
	)
	slideInTween:Play()

	local progressConnection
	progressConnection = RunService.RenderStepped:Connect(function()
		if notificationTimers[notificationId] == nil then
			progressConnection:Disconnect()
			return
		end

		local elapsed = tick() - notificationTimers[notificationId].startTime
		local progress = math.min(elapsed / notificationTimers[notificationId].duration, 1)

		if notificationGui and notificationGui.Parent then
			progressLine.Size = UDim2.new(totalWidth * progress, rpx(0), 0.045, rpx(0))
		end

		if progress >= 1 then
			progressConnection:Disconnect()

			if notificationGui and notificationGui.Parent then
				local slideOutTween = TweenService:Create(
					notificationGui,
					TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In),
					{Position = UDim2.new(0.99, rpx(0), 0.8, rpx(0))}
				)
				slideOutTween:Play()

				slideOutTween.Completed:Connect(function()
					if notificationGui then
						notificationGui:Destroy()
					end
					if notificationScreenGui then
						notificationScreenGui:Destroy()
					end
					notificationTimers[notificationId] = nil
				end)
			end
		end
	end)
end

local function createDebounce(key, duration)
    if not optionStates["_debounce"] then optionStates["_debounce"] = {} end
    if optionStates["_debounce"][key] then return false end
    optionStates["_debounce"][key] = true
    task.delay(duration or 0.05, function() optionStates["_debounce"][key] = nil end)
    return true
end

local function isPointInsideGui(guiObject, position)
    if not guiObject or not guiObject.Parent or not guiObject.Visible then
        return false
    end

    local absolutePosition = guiObject.AbsolutePosition
    local absoluteSize = guiObject.AbsoluteSize
    return position.X >= absolutePosition.X and position.X <= absolutePosition.X + absoluteSize.X and position.Y >= absolutePosition.Y and position.Y <= absolutePosition.Y + absoluteSize.Y
end

local function trainSpeedCFrameEquals(cf1, cf2)
    local posDif = (cf1.Position - cf2.Position).Magnitude
    local _, _, _, r00, r01, r02, r10, r11, r12, r20, r21, r22 = cf1:GetComponents()
    local _, _, _, s00, s01, s02, s10, s11, s12, s20, s21, s22 = cf2:GetComponents()
    if posDif > trainSpeedPrecision then return false end
    return math.abs(r00 - s00) < trainSpeedPrecision and math.abs(r01 - s01) < trainSpeedPrecision and math.abs(r02 - s02) < trainSpeedPrecision
       and math.abs(r10 - s10) < trainSpeedPrecision and math.abs(r11 - s11) < trainSpeedPrecision and math.abs(r12 - s12) < trainSpeedPrecision
       and math.abs(r20 - s20) < trainSpeedPrecision and math.abs(r21 - s21) < trainSpeedPrecision and math.abs(r22 - s22) < trainSpeedPrecision
end

local function getTrainSpeedCFrame(conveyor)
    local yOffset = (conveyor.Size.Y * 0.5) + 3
    local position = conveyor.Position + conveyor.CFrame.UpVector * yOffset
    return CFrame.new(position, position + conveyor.CFrame.LookVector)
end

local function prepareTrainSpeedRoot(hrp)
    if trainSpeedFrozenRoot ~= hrp then
        if trainSpeedFrozenRoot and trainSpeedFrozenRoot.Parent and trainSpeedOriginalAnchored ~= nil then
            trainSpeedFrozenRoot.Anchored = trainSpeedOriginalAnchored
        end
        trainSpeedFrozenRoot = hrp
        trainSpeedOriginalAnchored = hrp.Anchored
    end
    if hrp.Anchored then
        hrp.Anchored = false
    end
    hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
end

local function teleportToTrainSpeedStart(hrp)
    prepareTrainSpeedRoot(hrp)
    hrp.CFrame = trainSpeedStartCFrame
    hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
end

local function pullTrainSpeedRoot(hrp, targetCFrame, deltaTime)
    prepareTrainSpeedRoot(hrp)
    local currentPosition = hrp.Position
    local targetPosition = targetCFrame.Position
    local direction = targetPosition - currentPosition
    local distance = direction.Magnitude
    if distance <= trainSpeedArriveDistance then
        hrp.AssemblyLinearVelocity = Vector3.new(0, hrp.AssemblyLinearVelocity.Y, 0)
        return true
    end
    local safeDelta = math.max(deltaTime, 0.03)
    local stepDistance = math.min(distance, trainSpeedPullSpeed * safeDelta)
    local moveDirection = direction.Unit
    local newPosition = currentPosition + moveDirection * stepDistance
    hrp.CFrame = CFrame.new(newPosition, newPosition + targetCFrame.LookVector)
    local velocity = moveDirection * math.min(trainSpeedPullSpeed, distance / safeDelta)
    hrp.AssemblyLinearVelocity = Vector3.new(velocity.X, math.min(hrp.AssemblyLinearVelocity.Y, 0), velocity.Z)
    return distance - stepDistance <= trainSpeedArriveDistance
end

local function getTrainSpeedAreaObject(conveyor)
    local current = conveyor.Parent
    while current and current ~= workspace do
        if current:IsA("Model") and (current.Name == trainSpeedAdminModelName or current.Name == trainSpeedHourlyModelName) then
            return current
        end
        current = current.Parent
    end
    current = conveyor.Parent
    while current and current ~= workspace do
        if current:IsA("Model") then
            return current
        end
        current = current.Parent
    end
    return conveyor
end

local function createTrainSpeedArea(conveyor)
    local areaObject = getTrainSpeedAreaObject(conveyor)
    local areaCFrame = conveyor.CFrame
    local areaSize = conveyor.Size
    if areaObject and areaObject:IsA("Model") then
        local success, boundingCFrame, boundingSize = pcall(function()
            return areaObject:GetBoundingBox()
        end)
        if success and boundingCFrame and boundingSize then
            areaCFrame = boundingCFrame
            areaSize = boundingSize
        end
    end
    return {
        conveyor = conveyor,
        cframe = areaCFrame,
        size = areaSize,
        targetCFrame = getTrainSpeedCFrame(conveyor)
    }
end

local function isTrainSpeedInsideArea(hrp, area)
    if not area then
        return false
    end
    local localPosition = area.cframe:PointToObjectSpace(hrp.Position)
    return math.abs(localPosition.X) <= area.size.X * 0.5 + trainSpeedAreaMargin
        and math.abs(localPosition.Z) <= area.size.Z * 0.5 + trainSpeedAreaMargin
        and math.abs(localPosition.Y) <= area.size.Y * 0.5 + trainSpeedHeightMargin
end

local function pullTrainSpeedToArea(hrp, area, currentTrainSpeedSession)
    local arrived = false
    local lastPullTime = os.clock()
    while autoTrainSpeedActive and trainSpeedSessionId == currentTrainSpeedSession and not arrived do
        if not hrp.Parent then
            return false
        end
        local now = os.clock()
        arrived = pullTrainSpeedRoot(hrp, area.targetCFrame, now - lastPullTime)
        lastPullTime = now
        task.wait(0.03)
    end
    return arrived
end

local function startTrainSpeedMod()
    if trainSpeedLoopActive then
        return
    end
    autoTrainSpeedActive = true
    trainSpeedLoopActive = true
    trainSpeedSessionId = trainSpeedSessionId + 1
    local currentTrainSpeedSession = trainSpeedSessionId
    trainSpeedLastTime = 0

    task.spawn(function()
        while autoTrainSpeedActive and trainSpeedSessionId == currentTrainSpeedSession do
            local char, hum = getCharacterParts()
            if char and hum and hum.Health > 0 then
                char:SetAttribute("IsOnTreadmill", true)
                local waitTime = 0.25
                local config = getTrainSpeedConfig()
                if config and config.XP_TIME_BASED and config.XP_TIME_BASED.MIN_COOLDOWN then
                    waitTime = config.XP_TIME_BASED.MIN_COOLDOWN
                end
                local now = os.clock()
                if now - trainSpeedLastTime >= waitTime then
                    trainSpeedLastTime = now
                    local updateRemote = getTrainSpeedRemote()
                    if updateRemote then
                        pcall(function()
                            updateRemote:FireServer("Walking")
                        end)
                    end
                end
            end
            task.wait(0.05)
        end
    end)

    task.spawn(function()
        local startTeleported = false
        while autoTrainSpeedActive and trainSpeedSessionId == currentTrainSpeedSession do
            local char, hum, hrp = getCharacterParts()
            if char and hum and hrp and hum.Health > 0 then
                hum.PlatformStand = false
                if not startTeleported then
                    startTeleported = true
                    trainSpeedCurrentConveyor = nil
                    trainSpeedCurrentArea = nil
                    teleportToTrainSpeedStart(hrp)
                    task.wait(0.2)
                else
                    local conveyor = getBestTrainSpeedConveyor()
                    if conveyor then
                        local area = updateTrainSpeedTargetArea(conveyor)
                        if area then
                            if not isTrainSpeedInsideArea(hrp, area) then
                                pullTrainSpeedToArea(hrp, area, currentTrainSpeedSession)
                            end
                        end
                    end
                    task.wait(0.45)
                end
            else
                task.wait(0.35)
            end
        end
        if trainSpeedSessionId == currentTrainSpeedSession then
            trainSpeedLoopActive = false
        end
    end)

updateSearchFilter = function()
    local searchText = SearchTextBox.Text:lower()
    local isSearching = searchText ~= ""

    for _, element in pairs(MainContent:GetChildren()) do
        if element:IsA("GuiObject") and element.Name ~= "UIListLayout" then
            local tabName = optionTabMap[element]
            local elementName = optionElementsMap[element] or ""
            local headerKey = categoryHeaderKeys[element]
            local categoryKey = elementCategoryMap[element]

            if isSearching then
                if headerKey then
                    element.Visible = false
                elseif searchOnlyElements[element] then
                    element.Visible = elementName:lower():find(searchText, 1, true) ~= nil
                else
                    element.Visible = elementName ~= "" and elementName:lower():find(searchText, 1, true) ~= nil
                end
            elseif tabName == activeTab then
                if headerKey then
                    element.Visible = true
                elseif searchOnlyElements[element] then
                    element.Visible = false
                elseif categoryKey then
                    if elementName == "" then
                        element.Visible = categoryStates[categoryKey] == true and (element.Size.Y.Offset > 0 or element.Size.Y.Scale > 0)
                    else
                        element.Visible = categoryStates[categoryKey] == true
                    end
                else
                    element.Visible = true
                end
            else
                element.Visible = false
            end
        end
    end

    refreshMainCanvas()
end

local function startFly()
    local char = LP.Character or LP.CharacterAdded:Wait()
    local humanoid = char:FindFirstChildOfClass("Humanoid") or char:WaitForChild("Humanoid", 10)
    local root = char:FindFirstChild("HumanoidRootPart") or char:WaitForChild("HumanoidRootPart", 10)
    if not humanoid or not root then
        return
    end
    if isFlyActive and flyRoot == root and bg and bv and bg.Parent and bv.Parent then
        return
    end
    destroyFlyBodies(root)
    if optionStates["Auto Train Speed"] == true or autoTrainSpeedActive then
        stopTrainSpeedMod(true)
    end
    isFlyActive = true
    flyRoot = root
    flyHumanoid = humanoid
    root.Anchored = false
    humanoid.PlatformStand = false
    humanoid.AutoRotate = false
    pcall(function() humanoid:SetStateEnabled(Enum.HumanoidStateType.Freefall, false) end)
    pcall(function() humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false) end)
    pcall(function() humanoid:SetStateEnabled(Enum.HumanoidStateType.Landed, false) end)
    pcall(function() humanoid:ChangeState(Enum.HumanoidStateType.Physics) end)
    setFlyCollisions(char, true)
    bv = Instance.new("BodyVelocity")
    bv.Name = "FlyVel"
    bv.Velocity = Vector3.new(0, 0, 0)
    bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    bv.Parent = root
    bg = Instance.new("BodyGyro")
    bg.Name = "FlyGyro"
    bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bg.P = 9e4
    bg.CFrame = root.CFrame
    bg.Parent = root
    flyRenderConnection = RunService.RenderStepped:Connect(function()
        if not isFlyActive or not bv or not bg or not root.Parent or not humanoid.Parent then
            stopFly()
            return
        end
        root.Anchored = false
        humanoid.PlatformStand = false
        local cam = workspace.CurrentCamera
        if not cam then
            bv.Velocity = Vector3.new(0, 0, 0)
            return
        end
        local moveDir = humanoid.MoveDirection
        local velocity = Vector3.new(0, 0, 0)
        if moveDir.Magnitude > 0.05 then
            local camLook = cam.CFrame.LookVector
            local camRight = cam.CFrame.RightVector
            local flatLook = Vector3.new(camLook.X, 0, camLook.Z)
            local flatRight = Vector3.new(camRight.X, 0, camRight.Z)
            if flatLook.Magnitude > 0.001 then
                flatLook = flatLook.Unit
            else
                flatLook = Vector3.new(root.CFrame.LookVector.X, 0, root.CFrame.LookVector.Z)
                if flatLook.Magnitude > 0.001 then
                    flatLook = flatLook.Unit
                else
                    flatLook = Vector3.new(0, 0, -1)
                end
            end
            if flatRight.Magnitude > 0.001 then
                flatRight = flatRight.Unit
            else
                flatRight = Vector3.new(root.CFrame.RightVector.X, 0, root.CFrame.RightVector.Z)
                if flatRight.Magnitude > 0.001 then
                    flatRight = flatRight.Unit
                else
                    flatRight = Vector3.new(1, 0, 0)
                end
            end
            local forwardInput = moveDir:Dot(flatLook)
            local rightInput = moveDir:Dot(flatRight)
            local flyDirection = camLook * forwardInput + camRight * rightInput
            if flyDirection.Magnitude > 0.001 then
                velocity = flyDirection.Unit * flySpeed
            end
        end
        bv.Velocity = velocity
        bg.CFrame = cam.CFrame
    end)
end

            if wasFlyActive and optionStates["Auto Train Speed"] == true then
                task.spawn(startTrainSpeedMod)
            end
        end

        local noclipVal = optionStates["Noclip"]
        if type(noclipVal) == "function" then noclipVal = noclipVal() end
        if noclipVal then
            noclipEnabled = true
        else
            noclipEnabled = false
            for part, canCollide in pairs(noclipParts) do
                if part and part.Parent then
                    part.CanCollide = canCollide
                end
            end
            noclipParts = {}
        end
    end
end)

if type(hookmetamethod) == "function" and type(getnamecallmethod) == "function" then
    local oldNamecall
    oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
        local antiCheat = optionStates["Anti Cheat"]
        if type(antiCheat) == "function" then antiCheat = antiCheat() end
        if antiCheat then
            local method = getnamecallmethod()
            if method == "FireServer" and tostring(self) == "Cheat" then
                return nil
            end
        end
        return oldNamecall(self, ...)
    end)
end

local function createShiftLockUI()
    if shiftLockScreenGui then return end

    shiftLockScreenGui = Instance.new("ScreenGui")
    shiftLockScreenGui.Name = "ShiftLockScreenGui"
    shiftLockScreenGui.ResetOnSpawn = false
    shiftLockScreenGui.DisplayOrder = 999998
    shiftLockScreenGui.Parent = PlayerGui

    shiftLockIcon = Instance.new("ImageLabel")
    shiftLockIcon.Name = "ShiftLockIcon"
    shiftLockIcon.Size = UDim2.new(0.05, 0, 0.1, 0)
    shiftLockIcon.Position = UDim2.new(0.95, 0, 0.68, 0)
    shiftLockIcon.BackgroundTransparency = 1
    shiftLockIcon.Image = "rbxassetid://105987953182009"
    shiftLockIcon.Parent = shiftLockScreenGui
    shiftLockIcon.ZIndex = 999998

    shiftLockIcon.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            shiftLockActive = not shiftLockActive
            shiftLockIcon.Image = shiftLockActive and "rbxassetid://139211296111194" or "rbxassetid://105987953182009"
        end
    end)
end

local function destroyShiftLockUI()
    if shiftLockScreenGui then
        shiftLockScreenGui:Destroy()
        shiftLockScreenGui = nil
        shiftLockIcon = nil
    end
end

task.spawn(function()
    while task.wait(0.1) do
        local shiftLockEnabled = optionStates["Enable Shift Lock"]
        if type(shiftLockEnabled) == "function" then shiftLockEnabled = shiftLockEnabled() end

        if shiftLockEnabled and not shiftLockScreenGui then
            createShiftLockUI()
        elseif not shiftLockEnabled and shiftLockScreenGui then
            destroyShiftLockUI()
            shiftLockActive = false
            if shiftLockConnection then
                shiftLockConnection:Disconnect()
                shiftLockConnection = nil
            end
        end

        if shiftLockScreenGui and shiftLockActive then
            local char = LP.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local root = char.HumanoidRootPart
                local humanoid = char:FindFirstChildOfClass("Humanoid")
                if humanoid then
                    humanoid.AutoRotate = false
                    if not shiftLockConnection then
                        shiftLockConnection = RunService.RenderStepped:Connect(function()
                            local currentChar = LP.Character
                            if currentChar and currentChar:FindFirstChild("HumanoidRootPart") then
                                local currentRoot = currentChar.HumanoidRootPart
                                local camera = workspace.CurrentCamera
                                if camera then
                                    local lookVector = camera.CFrame.LookVector
                                    currentRoot.CFrame = CFrame.new(currentRoot.Position, currentRoot.Position + Vector3.new(lookVector.X, 0, lookVector.Z))
                                end
                            end
                        end)
                    end
                end
            end
        elseif shiftLockScreenGui and not shiftLockActive then
            if shiftLockConnection then
                shiftLockConnection:Disconnect()
                shiftLockConnection = nil
            end

            local char = LP.Character
            if char then
                local humanoid = char:FindFirstChildOfClass("Humanoid")
                if humanoid then
                    humanoid.AutoRotate = true
                end
            end
        end
    end
end)

Player.CharacterAdded:Connect(function()
    if shiftLockConnection then
        shiftLockConnection:Disconnect()
        shiftLockConnection = nil
    end
    local flyShouldRestart = optionStates["Fly"] == true or isFlyActive
    stopFly()
    noclipParts = {}
    local trainSpeedShouldRestart = (optionStates["Auto Train Speed"] == true or autoTrainSpeedActive) and not flyShouldRestart
    stopTrainSpeedMod(trainSpeedShouldRestart or flyShouldRestart)
    if flyShouldRestart then
        task.delay(0.35, function()
            if optionStates["Fly"] == true then
                startFly()
            end
        end)
    elseif trainSpeedShouldRestart then
        task.delay(0.6, function()
            if optionStates["Auto Train Speed"] == true then
                startTrainSpeedMod()
            end
        end)
    end
    if optionStates["Enable Speed"] == true then
        task.delay(0.15, function()
            if optionStates["Enable Speed"] == true then
                activateSpeedLock()
            end
        end)
    else
        deactivateSpeedLock()
    end
    autoCollectOrbActive = false
    collectSpecialKeyLastPosition = nil
end)

local function stopAllScriptFeatures()
    autoRebirthActive = false
    autoFarmTrophyActive = false
    autoBuyItemsActive = false
    autoCollectOrbActive = false
    stopCollectSpecialKeyLoop()
    stopTrainSpeedMod()
    deactivateSpeedLock()
    stopFly()
    noclipEnabled = false
    getgenv().ShiftLockEnabled = false

    for name, value in pairs(optionStates) do
        if type(value) == "boolean" then
            optionStates[name] = false
        end
    end

    for part, canCollide in pairs(noclipParts) do
        if part and part.Parent then
            pcall(function() part.CanCollide = canCollide end)
        end
    end
    noclipParts = {}

    if destroyShiftLockUI then
        destroyShiftLockUI()
    end
end
